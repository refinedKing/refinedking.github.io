类副构造

class hello(val msg: String) {
    private var num_ : Int = 0
    private var name_ : String = _

    def this(msg: String, num: Int) {
        this(msg)
        num_ = num
    }

    def this(msg: String, num: Int, name: String) {
        this(msg)
        num_ = num
        name_ = name
    }

    override def toString() : String = {
        name + msg + num
    }
}
